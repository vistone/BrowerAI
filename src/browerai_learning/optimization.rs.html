1/// Self-optimization system for autonomous improvement
2///
3/// Automatically adjusts system parameters and selects best-performing models
4use serde::{Deserialize, Serialize};
5use std::collections::HashMap;
6
7/// Optimization strategy
8#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
9pub enum OptimizationStrategy {
10    /// Optimize for speed
11    Speed,
12    /// Optimize for accuracy
13    Accuracy,
14    /// Optimize for memory usage
15    Memory,
16    /// Balance between speed and accuracy
17    Balanced,
18    /// Custom optimization target
19    Custom(String),
20}
21
22/// Configuration for self-optimization
23#[derive(Debug, Clone, Serialize, Deserialize)]
24pub struct OptimizationConfig {
25    /// Primary optimization strategy
26    pub strategy: OptimizationStrategy,
27    /// Minimum improvement threshold (0.0 to 1.0)
28    pub min_improvement_threshold: f32,
29    /// Enable automatic model switching
30    pub auto_switch_models: bool,
31    /// Enable parameter tuning
32    pub auto_tune_parameters: bool,
33}
34
35impl Default for OptimizationConfig {
36    fn default() -> Self {
37        Self {
38            strategy: OptimizationStrategy::Balanced,
39            min_improvement_threshold: 0.05, // 5% improvement
40            auto_switch_models: false,
41            auto_tune_parameters: false,
42        }
43    }
44}

46/// Performance measurement for optimization
47#[derive(Debug, Clone, Serialize, Deserialize)]
48pub struct PerformanceMeasurement {
49    /// Model or configuration ID
50    pub id: String,
51    /// Speed metric (lower is better)
52    pub speed_ms: f32,
53    /// Accuracy metric (higher is better)
54    pub accuracy: f32,
55    /// Memory usage in MB
56    pub memory_mb: f32,
57    /// Sample count
58    pub sample_count: usize,
59}
60
61impl PerformanceMeasurement {
62    pub fn new(id: impl Into<String>) -> Self {
63        Self {
64            id: id.into(),
65            speed_ms: 0.0,
66            accuracy: 0.0,
67            memory_mb: 0.0,
68            sample_count: 0,
69        }
70    }
71
72    /// Update metrics with new measurements
73    pub fn update(&mut self, speed_ms: f32, accuracy: f32, memory_mb: f32) {
74        let n = self.sample_count as f32;
75        self.speed_ms = (self.speed_ms * n + speed_ms) / (n + 1.0);
76        self.accuracy = (self.accuracy * n + accuracy) / (n + 1.0);
77        self.memory_mb = (self.memory_mb * n + memory_mb) / (n + 1.0);
78        self.sample_count += 1;
79    }
80
81    /// Calculate score based on strategy
82    pub fn score(&self, strategy: &OptimizationStrategy) -> f32 {
83        match strategy {
84            OptimizationStrategy::Speed => 1000.0 / self.speed_ms.max(1.0),
85            OptimizationStrategy::Accuracy => self.accuracy,
86            OptimizationStrategy::Memory => 1000.0 / self.memory_mb.max(1.0),
87            OptimizationStrategy::Balanced => {
88                // Balanced: 40% accuracy, 40% speed, 20% memory
89                0.4 * self.accuracy
90                    + 0.4 * (100.0 / self.speed_ms.max(1.0))
91                    + 0.2 * (100.0 / self.memory_mb.max(10.0))
92            }
93            OptimizationStrategy::Custom(_) => self.accuracy,
94        }
95    }
96}

98/// Self-optimizer that improves system performance
99pub struct SelfOptimizer {
100    /// Optimization configuration
101    config: OptimizationConfig,
102    /// Performance measurements by model/config ID
103    measurements: HashMap<String, PerformanceMeasurement>,
104    /// Currently active model/config
105    active_id: Option<String>,
106    /// Optimization history
107    optimization_count: usize,
108}
109
110impl SelfOptimizer {
111    /// Create a new self-optimizer
112    pub fn new(config: OptimizationConfig) -> Self {
113        Self {
114            config,
115            measurements: HashMap::new(),
116            active_id: None,
117            optimization_count: 0,
118        }
119    }
120
121    /// Create with default configuration
122    pub fn with_defaults() -> Self {
123        Self::new(OptimizationConfig::default())
124    }
125
126    /// Record performance measurement
127    pub fn record_performance(
128        &mut self,
129        id: impl Into<String>,
130        speed_ms: f32,
131        accuracy: f32,
132        memory_mb: f32,
133    ) {
134        let id_str = id.into();
135        self.measurements
136            .entry(id_str.clone())
137            .or_insert_with(|| PerformanceMeasurement::new(&id_str))
138            .update(speed_ms, accuracy, memory_mb);
139    }
140
141    /// Get best performing model/config based on strategy
142    pub fn get_best_performer(&self) -> Option<&PerformanceMeasurement> {
143        self.measurements
144            .values()
145            .filter(|m| m.sample_count > 0)
146            .max_by(|a, b| {
147                let a_score = a.score(&self.config.strategy);
148                let b_score = b.score(&self.config.strategy);
149                // Use total_cmp to handle NaN values safely
150                a_score.total_cmp(&b_score)
151            })
152    }
153
154    /// Check if optimization is needed
155    pub fn should_optimize(&self) -> bool {
156        if self.measurements.len() < 2 {
157            return false;
158        }
159
160        let best = match self.get_best_performer() {
161            Some(b) => b,
162            None => return false,
163        };
164
165        // Check if current active is significantly worse than best
166        if let Some(active_id) = &self.active_id {
167            if let Some(active) = self.measurements.get(active_id) {
168                let active_score = active.score(&self.config.strategy);
169                let best_score = best.score(&self.config.strategy);
170
171                if best_score > 0.0 {
172                    let improvement = (best_score - active_score) / best_score;
173                    return improvement >= self.config.min_improvement_threshold;
174                }
175            }
176        }
177
178        true
179    }
180
181    /// Perform optimization (switch to best model)
182    pub fn optimize(&mut self) -> Option<String> {
183        if !self.should_optimize() {
184            return None;
185        }
186
187        let best = self.get_best_performer()?;
188        let best_id = best.id.clone();
189
190        if self.config.auto_switch_models {
191            self.active_id = Some(best_id.clone());
192            self.optimization_count += 1;
193        }
194
195        Some(best_id)
196    }
197
198    /// Set active model/config
199    pub fn set_active(&mut self, id: impl Into<String>) {
200        self.active_id = Some(id.into());
201    }
202
203    /// Get active model/config ID
204    pub fn get_active(&self) -> Option<&str> {
205        self.active_id.as_deref()
206    }
207
208    /// Get performance measurements
209    pub fn get_measurements(&self) -> &HashMap<String, PerformanceMeasurement> {
210        &self.measurements
211    }
212
213    /// Get optimization statistics
214    pub fn get_stats(&self) -> OptimizationStats {
215        OptimizationStats {
216            model_count: self.measurements.len(),
217            optimization_count: self.optimization_count,
218            active_model: self.active_id.clone(),
219            best_model: self.get_best_performer().map(|m| m.id.clone()),
220        }
221    }
222
223    /// Clear all measurements
224    pub fn clear(&mut self) {
225        self.measurements.clear();
226        self.active_id = None;
227    }
228}
