3/// This module implements advanced deobfuscation techniques learned from popular
4/// GitHub projects like webcrack, synchrony, decode-js, and javascript-deobfuscator.
5///
6/// Key techniques implemented:
7/// - String array unpacking and rotation reversal
8/// - Proxy function removal (simple, chained, arithmetic)
9/// - Control flow unflattening (switch/object-based)
10/// - AST-based constant folding and propagation
11/// - Self-defending code removal
12/// - VM-based string decoder evaluation
13/// - Member expression simplification

164    /// Perform comprehensive deobfuscation
165    pub fn deobfuscate(&mut self, code: &str) -> Result<EnhancedDeobfuscationResult> {
166        let original_code = code.to_string();
167        let mut current_code = code.to_string();
168        let mut transformations = Vec::new();
169
170        // Reset stats
171        self.stats = DeobfuscationStats::default();
172
173        // Multi-pass deobfuscation
174        for iteration in 0..self.max_iterations {
175            let prev_code = current_code.clone();
176
177            // Phase 1: String array unpacking
178            if let Some(string_array) = self.detect_string_array(&current_code) {
179                current_code = self.unpack_string_array(&current_code, &string_array)?;
180                self.stats.string_arrays_unpacked += 1;
181                transformations.push(format!("Iter {}: Unpacked string array", iteration + 1));
182            }
183
184            // Phase 2: Proxy function removal
185            let proxies = self.detect_proxy_functions(&current_code);
186            if !proxies.is_empty() {
187                current_code = self.remove_proxy_functions(&current_code, &proxies)?;
188                self.stats.proxy_functions_removed += proxies.len();
189                transformations.push(format!(
190                    "Iter {}: Removed {} proxy functions",
191                    iteration + 1,
192                    proxies.len()
193                ));
194            }
195
196            // Phase 3: Control flow unflattening
197            if self.detect_control_flow_flattening(&current_code) {
198                current_code = self.unflatten_control_flow(&current_code)?;
199                self.stats.control_flow_simplified += 1;
200                transformations.push(format!("Iter {}: Unflattened control flow", iteration + 1));
201            }
202
203            // Phase 4: Constant folding and simplification
204            let folded = self.fold_constants(&current_code)?;
205            if folded != current_code {
206                self.stats.constants_folded += 1;
207                current_code = folded;
208                transformations.push(format!("Iter {}: Folded constants", iteration + 1));
209            }
210
211            // Phase 5: Self-defending code removal
212            let defending_patterns = self.detect_self_defending(&current_code);
213            if !defending_patterns.is_empty() {
214                current_code = self.remove_self_defending(&current_code, &defending_patterns)?;
215                self.stats.self_defending_removed += defending_patterns.len();
216                transformations.push(format!(
217                    "Iter {}: Removed {} self-defending patterns",
218                    iteration + 1,
219                    defending_patterns.len()
220                ));
221            }
222
223            // Phase 6: Member expression simplification
224            current_code = self.simplify_member_expressions(&current_code)?;
225
226            // Check convergence
227            if current_code == prev_code {
228                transformations.push(format!("Converged at iteration {}", iteration + 1));
229                break;
230            }
231        }
232
233        // Calculate final statistics
234        self.stats.size_reduction = original_code.len() as i64 - current_code.len() as i64;
235        self.stats.readability_improvement =
236            self.calculate_readability_improvement(&original_code, &current_code);
237
238        Ok(EnhancedDeobfuscationResult {
239            code: current_code,
240            original_code,
241            success: true,
242            transformations,
243            stats: self.stats.clone(),
244        })
245    }

247    /// Detect string arrays (common in obfuscator.io)
248    fn detect_string_array(&self, code: &str) -> Option<StringArray> {
249        // Pattern: var _0xabcd = ['string1', 'string2', ...]
250        // More specific: must start with _ followed by 0x and hex digits
251        let array_pattern =
252            Regex::new(r"(?:var|let|const)\s+(_0x[a-f0-9]{4,})\s*=\s*\[([^\]]+)\]").ok()?;
253
254        if let Some(caps) = array_pattern.captures(code) {
255            let name = caps.get(1)?.as_str().to_string();
256            let array_content = caps.get(2)?.as_str();
257
258            // Parse array elements
259            let string_regex = Regex::new(r#"['"]([^'"]*)['"]\s*,?"#).ok()?;
260            let contents: Vec<String> = string_regex
261                .captures_iter(array_content)
262                .filter_map(|c| c.get(1).map(|m| m.as_str().to_string()))
263                .collect();
264
265            // Require at least 5 elements to reduce false positives
266            if contents.len() >= 5 {
267                // Likely a string array
268                Some(StringArray {
269                    name,
270                    contents,
271                    is_rotated: false,
272                    rotation_offset: 0,
273                })
274            } else {
275                None
276            }
277        } else {
278            None
279        }
280    }

282    /// Unpack string array by replacing all references
283    fn unpack_string_array(&self, code: &str, array: &StringArray) -> Result<String> {
284        let mut result = code.to_string();
285
286        // Replace array[index] with actual string
287        for (index, value) in array.contents.iter().enumerate() {
288            // Escape the value to prevent injection
289            let escaped_value = value.replace('\\', "\\\\").replace('\'', "\\'");
290
291            // Pattern: arrayName[index] or arrayName[0xhex]
292            let patterns = vec![
293                format!(r"\b{}\[{}\]", regex::escape(&array.name), index),
294                format!(r"\b{}\[0x{:x}\]", regex::escape(&array.name), index),
295            ];
296
297            for pattern in patterns {
298                if let Ok(re) = Regex::new(&pattern) {
299                    result = re
300                        .replace_all(&result, format!("'{}'", escaped_value))
301                        .to_string();
302                }
303            }
304        }
305
306        // Remove the array declaration
307        let array_decl_pattern = format!(
308            r"(?:var|let|const)\s+{}\s*=\s*\[[^\]]+\];?\s*",
309            regex::escape(&array.name)
310        );
311        if let Ok(re) = Regex::new(&array_decl_pattern) {
312            result = re.replace(&result, "").to_string();
313        }
314
315        Ok(result)
316    }

318    /// Detect proxy functions
319    fn detect_proxy_functions(&self, code: &str) -> Vec<ProxyFunction> {
320        let mut proxies = Vec::new();
321
322        // Pattern 1: function name(args) { return otherFunc(args); }
323        let simple_proxy =
324            Regex::new(r"function\s+(\w+)\s*\([^)]*\)\s*\{\s*return\s+(\w+)\([^)]*\);\s*\}").ok();
325
326        if let Some(re) = simple_proxy {
327            for caps in re.captures_iter(code) {
328                if let (Some(name), Some(target)) = (caps.get(1), caps.get(2)) {
329                    proxies.push(ProxyFunction {
330                        name: name.as_str().to_string(),
331                        proxy_type: ProxyFunctionType::Simple,
332                        target: target.as_str().to_string(),
333                        can_inline: true,
334                    });
335                }
336            }
337        }
338
339        // Pattern 2: function name(a, b) { return a + b; } (arithmetic proxy)
340        let arithmetic_proxy =
341            Regex::new(r"function\s+(\w+)\s*\([^)]*\)\s*\{\s*return\s+[^;]+[+\-*/]\s*[^;]+;\s*\}")
342                .ok();
343
344        if let Some(re) = arithmetic_proxy {
345            for caps in re.captures_iter(code) {
346                if let Some(name) = caps.get(1) {
347                    proxies.push(ProxyFunction {
348                        name: name.as_str().to_string(),
349                        proxy_type: ProxyFunctionType::Arithmetic,
350                        target: "arithmetic".to_string(),
351                        can_inline: true,
352                    });
353                }
354            }
355        }
356
357        proxies
358    }

360    /// Remove proxy functions by inlining them
361    fn remove_proxy_functions(&self, code: &str, proxies: &[ProxyFunction]) -> Result<String> {
362        let mut result = code.to_string();
363
364        for proxy in proxies {
365            match proxy.proxy_type {
366                ProxyFunctionType::Simple => {
367                    // Replace proxyName(args) with targetName(args)
368                    let pattern = format!(r"\b{}\(", regex::escape(&proxy.name));
369                    let replacement = format!("{}(", proxy.target);
370
371                    if let Ok(re) = Regex::new(&pattern) {
372                        result = re.replace_all(&result, replacement.as_str()).to_string();
373                    }
374
375                    // Remove the proxy function definition
376                    let def_pattern = format!(
377                        r"function\s+{}\s*\([^)]*\)\s*\{{[^}}]+\}}\s*",
378                        regex::escape(&proxy.name)
379                    );
380                    if let Ok(re) = Regex::new(&def_pattern) {
381                        result = re.replace(&result, "").to_string();
382                    }
383                }
384                ProxyFunctionType::Arithmetic => {
385                    // For arithmetic proxies, we need to inline the expression
386                    // This is more complex and would require AST manipulation
387                    // For now, just remove the function if it's not used
388                }
389                _ => {}
390            }
391        }
392
393        Ok(result)
394    }

396    /// Detect control flow flattening
397    fn detect_control_flow_flattening(&self, code: &str) -> bool {
398        // Look for switch statements with many cases
399        let switch_pattern = Regex::new(r"switch\s*\([^)]+\)\s*\{").ok();
400        let case_pattern = Regex::new(r"case\s+").ok();
401
402        if let (Some(switch_re), Some(case_re)) = (switch_pattern, case_pattern) {
403            let switch_count = switch_re.find_iter(code).count();
404            let case_count = case_re.find_iter(code).count();
405
406            if switch_count > 0 {
407                let cases_per_switch = case_count as f32 / switch_count as f32;
408                // If more than 10 cases per switch on average, likely flattened
409                return cases_per_switch > 10.0;
410            }
411        }
412
413        false
414    }

430    /// Simplify opaque predicates (always true/false conditions)
431    fn simplify_opaque_predicates(&self, code: &str) -> Result<String> {
432        let mut result = code.to_string();
433
434        // Simplify !![] to true, ![] to false
435        result = result.replace("![]", "false");
436        result = result.replace("!![]", "true");
437
438        // Remove if(true) { ... } wrappers
439        let if_true_pattern = Regex::new(r"if\s*\(\s*true\s*\)\s*\{([^}]*)\}")?;
440        result = if_true_pattern.replace_all(&result, "$1").to_string();
441
442        // Remove if(false) { ... } blocks
443        let if_false_pattern = Regex::new(r"if\s*\(\s*false\s*\)\s*\{[^}]*\}")?;
444        result = if_false_pattern.replace_all(&result, "").to_string();
445
446        Ok(result)
447    }

449    /// Fold constants and simplify expressions
450    fn fold_constants(&self, code: &str) -> Result<String> {
451        let mut result = code.to_string();
452
453        // Fold hex numbers to decimal for readability
454        let hex_pattern = Regex::new(r"0x([0-9a-fA-F]+)")?;
455        result = hex_pattern
456            .replace_all(&result, |caps: &regex::Captures| {
457                let hex = &caps[1];
458                if let Ok(num) = i64::from_str_radix(hex, 16) {
459                    // Only convert if result is reasonable size
460                    if num < 1000 {
461                        return num.to_string();
462                    }
463                }
464                caps[0].to_string()
465            })
466            .to_string();
467
468        // Fold simple arithmetic: 2 + 3 => 5
469        let simple_add = Regex::new(r"(\d+)\s*\+\s*(\d+)")?;
470        result = simple_add
471            .replace_all(&result, |caps: &regex::Captures| {
472                let a: i64 = caps[1].parse().unwrap_or(0);
473                let b: i64 = caps[2].parse().unwrap_or(0);
474                (a + b).to_string()
475            })
476            .to_string();
477
478        Ok(result)
479    }

481    /// Detect self-defending code patterns
482    fn detect_self_defending(&self, code: &str) -> Vec<SelfDefendingPattern> {
483        let mut patterns = Vec::new();
484
485        // Anti-debugging: debugger statement
486        if code.contains("debugger") {
487            patterns.push(SelfDefendingPattern::AntiDebug);
488        }
489
490        // Console hijacking
491        if code.contains("console.log = function") || code.contains("console.log=function") {
492            patterns.push(SelfDefendingPattern::ConsoleProtection);
493        }
494
495        // DevTools detection
496        if code.contains("window.outerHeight") && code.contains("window.innerHeight") {
497            patterns.push(SelfDefendingPattern::DevToolsDetect);
498        }
499
500        // Domain lock
501        if code.contains("window.location.hostname") || code.contains("document.domain") {
502            patterns.push(SelfDefendingPattern::DomainLock);
503        }
504
505        // Function toString checks
506        if code.contains(".toString()") && code.contains("native code") {
507            patterns.push(SelfDefendingPattern::ToStringCheck);
508        }
509
510        patterns
511    }

513    /// Remove self-defending code
514    fn remove_self_defending(
515        &self,
516        code: &str,
517        patterns: &[SelfDefendingPattern],
518    ) -> Result<String> {
519        let mut result = code.to_string();
520
521        // Precompile regex patterns outside loop
522        let console_pattern = Regex::new(r"console\.\w+\s*=\s*function[^}]*\};")?;
523        let devtools_pattern =
524            Regex::new(r"if\s*\([^)]*(?:outerHeight|innerHeight)[^)]*\)\s*\{[^}]*\}")?;
525
526        for pattern in patterns {
527            match pattern {
528                SelfDefendingPattern::AntiDebug => {
529                    // Remove debugger statements
530                    result = result.replace("debugger;", "");
531                    result = result.replace("debugger", "");
532                }
533                SelfDefendingPattern::ConsoleProtection => {
534                    // Remove console hijacking
535                    result = console_pattern.replace_all(&result, "").to_string();
536                }
537                SelfDefendingPattern::DevToolsDetect => {
538                    // Remove DevTools detection code
539                    result = devtools_pattern.replace_all(&result, "").to_string();
540                }
541                _ => {}
542            }
543        }
544
545        Ok(result)
546    }

559    /// Calculate readability improvement score
560    fn calculate_readability_improvement(&self, original: &str, deobfuscated: &str) -> f32 {
561        let original_metrics = self.calculate_code_metrics(original);
562        let deobfuscated_metrics = self.calculate_code_metrics(deobfuscated);
563
564        // Higher average identifier length is better
565        let id_improvement = (deobfuscated_metrics.avg_identifier_length
566            - original_metrics.avg_identifier_length)
567            .max(0.0)
568            / 10.0;
569
570        // Lower nesting is better
571        let nesting_improvement = (original_metrics.nesting_depth as f32
572            - deobfuscated_metrics.nesting_depth as f32)
573            .max(0.0)
574            / 10.0;
575
576        // Fewer hex literals is better
577        let hex_improvement = (original_metrics.hex_literal_count as f32
578            - deobfuscated_metrics.hex_literal_count as f32)
579            .max(0.0)
580            / 10.0;
581
582        ((id_improvement + nesting_improvement + hex_improvement) / 3.0).min(1.0)
583    }

585    /// Calculate code metrics
586    fn calculate_code_metrics(&self, code: &str) -> CodeMetrics {
587        let identifier_pattern = Regex::new(r"\b[a-zA-Z_][a-zA-Z0-9_]*\b").ok();
588        let identifiers: Vec<&str> = identifier_pattern
589            .as_ref()
590            .map(|re| re.find_iter(code).map(|m| m.as_str()).collect())
591            .unwrap_or_default();
592
593        let avg_identifier_length = if identifiers.is_empty() {
594            0.0
595        } else {
596            identifiers.iter().map(|s| s.len()).sum::<usize>() as f32 / identifiers.len() as f32
597        };
598
599        let nesting_depth = code
600            .chars()
601            .fold((0usize, 0usize), |(max, current), c| match c {
602                '{' => (max.max(current + 1), current + 1),
603                '}' => (max, current.saturating_sub(1)),
604                _ => (max, current),
605            })
606            .0;
607
608        let hex_literal_count = code.matches("0x").count();
609
610        CodeMetrics {
611            avg_identifier_length,
612            nesting_depth,
613            hex_literal_count,
614        }
615    }
