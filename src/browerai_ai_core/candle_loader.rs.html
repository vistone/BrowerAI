21/// Minimal GGUF loader helpers for Candle-backed LLMs (Qwen2/Qwen2.5 GGUF).
22///
23/// This is intentionally lightweight: we only validate files, pick a device,
24/// and rely on Candle's quantized Qwen2 loader to bring the weights in. Callers
25/// can wrap the returned `ModelWeights` into their own generation pipeline.
26#[cfg(feature = "candle")]
27pub struct CandleModelLoader;
28
29/// Simple wrapper around Candle quantized LLaMA/Qwen weights + tokenizer for generation.
30#[cfg(feature = "candle")]
31pub struct CandleCodeLlm {
32    model: ModelWeights,
33    tokenizer: Tokenizer,
34    device: Device,
35    eos_token: Option<u32>,
36}

38#[cfg(feature = "candle")]
39impl CandleModelLoader {
40    /// Pick a device. Prefer CUDA:0 if available, otherwise fall back to CPU.
41    pub fn device(prefer_gpu: bool) -> Device {
42        if prefer_gpu {
43            log::info!("Attempting to use CUDA GPU (device 0)...");
44            match Device::new_cuda(0) {
45                Ok(dev) => {
46                    log::info!("✓ Successfully using CUDA GPU");
47                    dev
48                }
49                Err(e) => {
50                    log::warn!("✗ CUDA initialization failed: {}. Falling back to CPU.", e);
51                    Device::Cpu
52                }
53            }
54        } else {
55            log::info!("Using CPU (PREFER_GPU=0 or not set)");
56            Device::Cpu
57        }
58    }
59
60    /// Load quantized GGUF weights (Qwen2-style) for code LLMs.
61    ///
62    /// Note: Qwen2.5-Coder-7B GGUF exposes `qwen2.*` metadata keys; we use the
63    /// dedicated quantized_qwen2 loader in Candle.
64    pub fn load_gguf_weights(model_path: &Path, device: &Device) -> Result<ModelWeights> {
65        if !model_path.exists() {
66            anyhow::bail!("Model file not found: {:?}", model_path);
67        }
68
69        let mut file = File::open(model_path)
70            .with_context(|| format!("Failed to open model file: {:?}", model_path))?;
71
72        let content = gguf_file::Content::read(&mut file)
73            .with_context(|| format!("Failed to read GGUF header from {:?}", model_path))?;
74
75        // Build weights from GGUF content + reader (new Candle API)
76        let weights = ModelWeights::from_gguf(content, &mut file, device)
77            .with_context(|| format!("Failed to load GGUF weights from {:?}", model_path))?;
78
79        Ok(weights)
80    }
81
82    /// Load a tokenizer JSON that accompanies the GGUF checkpoint.
83    pub fn load_tokenizer(tokenizer_path: &Path) -> Result<Tokenizer> {
84        if !tokenizer_path.exists() {
85            anyhow::bail!("Tokenizer file not found: {:?}", tokenizer_path);
86        }
87
88        Tokenizer::from_file(tokenizer_path)
89            .map_err(|e| anyhow::anyhow!("Failed to load tokenizer: {e}"))
90    }
91}

93#[cfg(feature = "candle")]
94impl CandleCodeLlm {
95    /// Build an inference-ready code LLM from GGUF weights and tokenizer.
96    pub fn new(model_path: &Path, tokenizer_path: &Path, prefer_gpu: bool) -> Result<Self> {
97        let device = CandleModelLoader::device(prefer_gpu);
98        log::info!("Device selected: {:?}", device);
99        log::info!("Loading GGUF weights from {:?}...", model_path);
100        let model = CandleModelLoader::load_gguf_weights(model_path, &device)?;
101        log::info!("Loading tokenizer from {:?}...", tokenizer_path);
102        let tokenizer = CandleModelLoader::load_tokenizer(tokenizer_path)?;
103
104        // Try common EOS tokens; fall back to tokenizer's default if present.
105        let vocab = tokenizer.get_vocab(true);
106        let eos_token = ["<|im_end|>", "<|endoftext|>"]
107            .iter()
108            .find_map(|tok| vocab.get(*tok).copied());
109
110        Ok(Self {
111            model,
112            tokenizer,
113            device,
114            eos_token,
115        })
116    }
117
118    /// Greedy / top-k / top-p generation (single batch) with repeat penalty.
119    #[allow(clippy::too_many_arguments)]
120    pub fn generate(
121        &mut self,
122        prompt: &str,
123        max_new_tokens: usize,
124        temperature: f64,
125        top_k: Option<usize>,
126        top_p: Option<f64>,
127        repeat_penalty: f64,
128        repeat_last_n: usize,
129    ) -> Result<String> {
130        let encoding = self
131            .tokenizer
132            .encode(prompt, true)
133            .map_err(|e| anyhow::anyhow!("tokenize failed: {e}"))?;
134        let mut tokens: Vec<u32> = encoding.get_ids().to_vec();
135        let prompt_len = tokens.len();
136
137        let sampling = if temperature <= 0.0 {
138            Sampling::ArgMax
139        } else {
140            match (top_k, top_p) {
141                (None, None) => Sampling::All { temperature },
142                (Some(k), None) => Sampling::TopK { k, temperature },
143                (None, Some(p)) => Sampling::TopP { p, temperature },
144                (Some(k), Some(p)) => Sampling::TopKThenTopP { k, p, temperature },
145            }
146        };
147        let mut logits_processor = LogitsProcessor::from_sampling(42, sampling);
148
149        // First step: feed full prompt to prime KV cache and sample next token
150        let input = candle_core::Tensor::new(tokens.as_slice(), &self.device)?.unsqueeze(0)?;
151        let logits = self.model.forward(&input, 0)?.squeeze(0)?;
152        let mut next_token = logits_processor.sample(&logits)?;
153        tokens.push(next_token);
154
155        // Generate remaining tokens
156        for idx in 0..max_new_tokens.saturating_sub(1) {
157            let input = candle_core::Tensor::new(&[next_token], &self.device)?.unsqueeze(0)?;
158            let mut logits = self.model.forward(&input, prompt_len + idx)?.squeeze(0)?;
159
160            if repeat_penalty != 1.0 {
161                let start_at = tokens.len().saturating_sub(repeat_last_n);
162                logits = apply_repeat_penalty(&logits, repeat_penalty as f32, &tokens[start_at..])?;
163            }
164
165            next_token = logits_processor.sample(&logits)?;
166            tokens.push(next_token);
167
168            if let Some(eos) = self.eos_token {
169                if next_token == eos {
170                    break;
171                }
172            }
173        }
174
175        // Decode full sequence; callers can trim the prompt prefix if需要
176        let text = self
177            .tokenizer
178            .decode(&tokens, true)
179            .map_err(|e| anyhow::anyhow!("decode failed: {e}"))?;
180        Ok(text)
181    }
182}
