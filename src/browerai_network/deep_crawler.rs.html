10/// Configuration for deep crawling
11#[derive(Debug, Clone)]
12pub struct CrawlConfig {
13    /// Maximum depth to crawl (0 = single page, 1 = page + direct links, etc.)
14    pub max_depth: usize,
15    /// Maximum number of pages to crawl
16    pub max_pages: usize,
17    /// Timeout for each request
18    pub request_timeout: Duration,
19    /// Delay between requests (politeness)
20    pub request_delay: Duration,
21    /// Follow external links
22    pub follow_external: bool,
23    /// Respect robots.txt
24    pub respect_robots: bool,
25    /// Maximum concurrent requests
26    pub max_concurrent: usize,
27}
28
29impl Default for CrawlConfig {
30    fn default() -> Self {
31        Self {
32            max_depth: 3,
33            max_pages: 50,
34            request_timeout: Duration::from_secs(30),
35            request_delay: Duration::from_millis(500),
36            follow_external: false,
37            respect_robots: true,
38            max_concurrent: 5,
39        }
40    }
41}

66/// Deep web crawler
67pub struct DeepCrawler {
68    config: CrawlConfig,
69    visited: HashSet<String>,
70    queue: VecDeque<(String, usize, Option<String>)>, // (URL, depth, parent)
71    pages: Vec<CrawledPage>,
72    site_map: HashMap<String, Vec<String>>,
73    errors: Vec<String>,
74    start_time: Instant,
75}
76
77impl DeepCrawler {
78    /// Create a new deep crawler with configuration
79    pub fn new(config: CrawlConfig) -> Self {
80        Self {
81            config,
82            visited: HashSet::new(),
83            queue: VecDeque::new(),
84            pages: Vec::new(),
85            site_map: HashMap::new(),
86            errors: Vec::new(),
87            start_time: Instant::now(),
88        }
89    }
90
91    /// Create a crawler with default configuration
92    pub fn with_defaults() -> Self {
93        Self::new(CrawlConfig::default())
94    }
95
96    /// Start crawling from a seed URL
97    pub fn crawl(&mut self, seed_url: &str) -> Result<CrawlResult> {
98        log::info!("Starting deep crawl from: {}", seed_url);
99        self.start_time = Instant::now();
100
101        // Validate and normalize seed URL
102        let base_url = Url::parse(seed_url).context("Invalid seed URL")?;
103
104        // Add seed to queue
105        self.queue.push_back((seed_url.to_string(), 0, None));
106
107        // Main crawl loop
108        while let Some((url, depth, parent)) = self.queue.pop_front() {
109            // Check limits
110            if self.pages.len() >= self.config.max_pages {
111                log::info!("Reached max pages limit: {}", self.config.max_pages);
112                break;
113            }
114
115            if depth > self.config.max_depth {
116                continue;
117            }
118
119            // Skip if already visited
120            if self.visited.contains(&url) {
121                continue;
122            }
123
124            // Mark as visited
125            self.visited.insert(url.clone());
126
127            // Crawl the page
128            match self.crawl_page(&url, depth, parent.clone(), &base_url) {
129                Ok(page) => {
130                    // Extract and queue links
131                    let links = page.links.clone();
132                    self.pages.push(page);
133
134                    // Add children to site map
135                    if let Some(parent_url) = &parent {
136                        self.site_map
137                            .entry(parent_url.clone())
138                            .or_default()
139                            .push(url.clone());
140                    }
141
142                    // Queue discovered links
143                    for link in links {
144                        if !self.visited.contains(&link) {
145                            self.queue.push_back((link, depth + 1, Some(url.clone())));
146                        }
147                    }
148                }
149                Err(e) => {
150                    log::warn!("Failed to crawl {}: {}", url, e);
151                    self.errors.push(format!("Failed to crawl {}: {}", url, e));
152                }
153            }
154
155            // Politeness delay
156            if !self.queue.is_empty() {
157                std::thread::sleep(self.config.request_delay);
158            }
159        }
160
161        let total_time = self.start_time.elapsed();
162        log::info!(
163            "Crawl complete: {} pages crawled, {} errors, {:?}",
164            self.pages.len(),
165            self.errors.len(),
166            total_time
167        );
168
169        Ok(CrawlResult {
170            pages: self.pages.clone(),
171            site_map: self.site_map.clone(),
172            total_time,
173            pages_crawled: self.pages.len(),
174            pages_skipped: self.visited.len() - self.pages.len(),
175            errors: self.errors.clone(),
176        })
177    }
178
179    /// Crawl a single page
180    fn crawl_page(
181        &self,
182        url: &str,
183        depth: usize,
184        parent: Option<String>,
185        base_url: &Url,
186    ) -> Result<CrawledPage> {
187        log::debug!("Crawling [depth={}]: {}", depth, url);
188        let page_start = Instant::now();
189
190        // Make HTTP request
191        let client = reqwest::blocking::Client::builder()
192            .timeout(self.config.request_timeout)
193            .user_agent("BrowerAI/0.1.0 DeepCrawler")
194            .build()?;
195
196        let response = client.get(url).send().context("HTTP request failed")?;
197
198        let status_code = response.status().as_u16();
199        let html_content = response.text().context("Failed to read response body")?;
200
201        // Extract links from HTML
202        let links = self.extract_links(&html_content, url, base_url)?;
203
204        let crawl_time = page_start.elapsed();
205
206        Ok(CrawledPage {
207            url: url.to_string(),
208            depth,
209            html_content,
210            links,
211            parent_url: parent,
212            crawl_time,
213            status_code,
214        })
215    }
216
217    /// Extract and normalize links from HTML
218    fn extract_links(&self, html: &str, current_url: &str, base_url: &Url) -> Result<Vec<String>> {
219        let mut links = Vec::new();
220        let current_parsed = Url::parse(current_url)?;
221
222        // Simple regex-based link extraction
223        // In production, use a proper HTML parser like scraper or html5ever
224        let link_regex = regex::Regex::new(r#"(?i)href=["']([^"']+)["']"#)?;
225
226        for cap in link_regex.captures_iter(html) {
227            if let Some(href) = cap.get(1) {
228                let href_str = href.as_str();
229
230                // Skip fragments and javascript links
231                if href_str.starts_with('#') || href_str.starts_with("javascript:") {
232                    continue;
233                }
234
235                // Resolve relative URLs
236                if let Ok(absolute_url) = current_parsed.join(href_str) {
237                    let url_str = absolute_url.to_string();
238
239                    // Check if we should follow this link
240                    if self.should_follow(&absolute_url, base_url) {
241                        links.push(url_str);
242                    }
243                }
244            }
245        }
246
247        // Deduplicate
248        links.sort();
249        links.dedup();
250
251        Ok(links)
252    }
253
254    /// Determine if a URL should be followed
255    fn should_follow(&self, url: &Url, base_url: &Url) -> bool {
256        // Skip non-HTTP(S) URLs
257        if url.scheme() != "http" && url.scheme() != "https" {
258            return false;
259        }
260
261        // Check external links
262        if !self.config.follow_external && url.host() != base_url.host() {
263            return false;
264        }
265
266        // Skip common non-content URLs
267        let path = url.path().to_lowercase();
268        let skip_extensions = [".pdf", ".jpg", ".jpeg", ".png", ".gif", ".zip", ".exe"];
269        for ext in &skip_extensions {
270            if path.ends_with(ext) {
271                return false;
272            }
273        }
274
275        true
276    }

279/// Analyze crawl results to understand site structure
280pub fn analyze_site_structure(result: &CrawlResult) -> SiteStructureAnalysis {
281    let mut analysis = SiteStructureAnalysis::default();
282
283    // Calculate depth distribution
284    for page in &result.pages {
285        *analysis.depth_distribution.entry(page.depth).or_insert(0) += 1;
286    }
287
288    // Find entry points (pages at depth 0 or 1)
289    analysis.entry_points = result
290        .pages
291        .iter()
292        .filter(|p| p.depth <= 1)
293        .map(|p| p.url.clone())
294        .collect();
295
296    // Calculate average links per page
297    let total_links: usize = result.pages.iter().map(|p| p.links.len()).sum();
298    analysis.avg_links_per_page = if !result.pages.is_empty() {
299        total_links as f64 / result.pages.len() as f64
300    } else {
301        0.0
302    };
303
304    // Identify hub pages (pages with many outgoing links)
305    let mut hub_candidates: Vec<_> = result
306        .pages
307        .iter()
308        .filter(|p| p.links.len() > 10)
309        .map(|p| (p.url.clone(), p.links.len()))
310        .collect();
311    hub_candidates.sort_by(|a, b| b.1.cmp(&a.1));
312    analysis.hub_pages = hub_candidates.into_iter().take(5).collect();
313
314    // Site map summary
315    analysis.total_pages = result.pages.len();
316    analysis.max_depth_reached = result.pages.iter().map(|p| p.depth).max().unwrap_or(0);
317
318    analysis
319}
