1//! 智能渲染器 - 渲染阶段
2//!
3//! 将生成的体验渲染给用户
4
5use super::generation::GeneratedExperience;
6use anyhow::Result;
7
8/// 智能渲染器
9pub struct IntelligentRenderer {
10    /// 当前选择的体验
11    current_experience: GeneratedExperience,
12
13    /// 所有可用体验
14    available_experiences: Vec<GeneratedExperience>,
15}
16
17/// 渲染结果
18pub struct RenderResult {
19    /// 最终HTML
20    pub final_html: String,
21
22    /// 最终CSS
23    pub final_css: String,
24
25    /// 最终JS（原始 + 桥接）
26    pub final_js: String,
27
28    /// 渲染统计
29    pub stats: RenderStats,
30}
31
32/// 渲染统计
33#[derive(Debug, Clone)]
34pub struct RenderStats {
35    pub html_size: usize,
36    pub css_size: usize,
37    pub js_size: usize,
38    pub functions_bridged: usize,
39}
40
41impl IntelligentRenderer {
42    /// 创建渲染器
43    pub fn new(
44        current_experience: GeneratedExperience,
45        available_experiences: Vec<GeneratedExperience>,
46    ) -> Self {
47        Self {
48            current_experience,
49            available_experiences,
50        }
51    }
52
53    /// 智能渲染
54    pub fn render(&self) -> Result<RenderResult> {
55        // 1. 组装最终页面
56        let final_html = self.assemble_page()?;
57
58        // 2. 收集统计信息
59        let stats = self.collect_stats()?;
60
61        Ok(RenderResult {
62            final_html: final_html.clone(),
63            final_css: self.current_experience.css.clone(),
64            final_js: self.current_experience.bridge_js.clone(),
65            stats,
66        })
67    }
68
69    /// 运行时切换体验
70    pub fn switch_experience(&mut self, variant_id: &str) -> Result<()> {
71        let experience = self
72            .available_experiences
73            .iter()
74            .find(|e| e.variant_id == variant_id)
75            .ok_or_else(|| anyhow::anyhow!("Experience not found"))?;
76
77        self.current_experience = (*experience).clone();
78        Ok(())
79    }
80
81    /// 获取可用体验列表
82    pub fn get_available_experiences(&self) -> Vec<String> {
83        self.available_experiences
84            .iter()
85            .map(|e| e.variant_id.clone())
86            .collect()
87    }
88
89    fn assemble_page(&self) -> Result<String> {
90        let mut page = self.current_experience.html.clone();
91
92        // 注入CSS
93        let css_tag = format!(
94            "<style>\n{}\n</style>\n</head>",
95            self.current_experience.css
96        );
97        page = page.replace("</head>", &css_tag);
98
99        // 注入JS
100        let js_tag = format!(
101            "<script>\n{}\n</script>\n</body>",
102            self.current_experience.bridge_js
103        );
104        page = page.replace("</body>", &js_tag);
105
106        // 注入体验切换器
107        let switcher = self.generate_experience_switcher();
108        page = page.replace("</body>", &format!("{}\n</body>", switcher));
109
110        Ok(page)
111    }
112
113    fn generate_experience_switcher(&self) -> String {
114        let mut switcher = String::from(
115            "<div id='browerai-switcher' style='position:fixed;bottom:20px;right:20px;background:#fff;padding:15px;border-radius:8px;box-shadow:0 4px 6px rgba(0,0,0,0.1);z-index:9999;'>\n"
116        );
117
118        switcher.push_str("  <h4 style='margin:0 0 10px 0;font-size:14px;'>体验切换</h4>\n");
119
120        for exp in &self.available_experiences {
121            let is_current = exp.variant_id == self.current_experience.variant_id;
122            let style = if is_current {
123                "background:#3B82F6;color:white;border:none;padding:8px 12px;margin:4px 0;cursor:pointer;border-radius:4px;display:block;width:100%;"
124            } else {
125                "background:#F3F4F6;color:#111827;border:none;padding:8px 12px;margin:4px 0;cursor:pointer;border-radius:4px;display:block;width:100%;"
126            };
127
128            switcher.push_str(&format!(
129                "  <button onclick=\"location.reload()\" style=\"{}\">{}</button>\n",
130                style, exp.variant_id
131            ));
132        }
133
134        switcher.push_str("</div>\n");
135
136        switcher
137    }
138
139    fn collect_stats(&self) -> Result<RenderStats> {
140        Ok(RenderStats {
141            html_size: self.current_experience.html.len(),
142            css_size: self.current_experience.css.len(),
143            js_size: self.current_experience.bridge_js.len(),
144            functions_bridged: self
145                .current_experience
146                .function_validation
147                .function_map
148                .len(),
149        })
150    }
151}

153#[cfg(test)]
154mod tests {
155    use super::*;
156    use crate::generation::IntelligentGeneration;
157    use crate::reasoning::IntelligentReasoning;
158    use crate::site_understanding::SiteUnderstanding;

160    #[test]
161    fn test_render_process() {
162        let html = "<html><body><h1>Test</h1></body></html>".to_string();
163        let css = "".to_string();
164        let js = "".to_string();
165
166        let understanding = SiteUnderstanding::learn_from_content(html, css, js).unwrap();
167        let reasoning = IntelligentReasoning::new(understanding);
168        let reasoning_result = reasoning.reason().unwrap();
169
170        let generation = IntelligentGeneration::new(reasoning_result);
171        let experiences = generation.generate().unwrap();
172
173        assert!(!experiences.is_empty());
174
175        let renderer = IntelligentRenderer::new(experiences[0].clone(), experiences.clone());
176
177        let result = renderer.render().unwrap();
178
179        assert!(!result.final_html.is_empty());
180        assert!(!result.final_css.is_empty());
181        assert!(!result.final_js.is_empty());
182        assert!(result.stats.html_size > 0);
183    }

185    #[test]
186    fn test_experience_switching() {
187        let html = "<html><body><h1>Test</h1></body></html>".to_string();
188        let css = "".to_string();
189        let js = "".to_string();
190
191        let understanding = SiteUnderstanding::learn_from_content(html, css, js).unwrap();
192        let reasoning = IntelligentReasoning::new(understanding);
193        let reasoning_result = reasoning.reason().unwrap();
194
195        let generation = IntelligentGeneration::new(reasoning_result);
196        let experiences = generation.generate().unwrap();
197
198        let mut renderer = IntelligentRenderer::new(experiences[0].clone(), experiences.clone());
199
200        let available = renderer.get_available_experiences();
201        assert!(available.len() >= 3);
202
203        // 切换到不同的体验

206        }
