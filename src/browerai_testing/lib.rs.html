14/// Test result for a single website test
15#[derive(Debug, Clone)]
16pub struct WebsiteTestResult {
17    /// URL that was tested
18    pub url: String,
19    /// Whether HTML parsing succeeded
20    pub html_parse_success: bool,
21    /// Whether CSS parsing succeeded
22    pub css_parse_success: bool,
23    /// Whether JS parsing succeeded  
24    pub js_parse_success: bool,
25    /// Time taken to parse HTML
26    pub html_parse_time: Duration,
27    /// Time taken to parse CSS
28    pub css_parse_time: Duration,
29    /// Time taken to parse JS
30    pub js_parse_time: Duration,
31    /// Total elements parsed
32    pub elements_parsed: usize,
33    /// Total CSS rules parsed
34    pub css_rules_parsed: usize,
35    /// Total JS statements parsed
36    pub js_statements_parsed: usize,
37    /// Any errors encountered
38    pub errors: Vec<String>,
39}
40
41impl WebsiteTestResult {
42    /// Create a new test result
43    pub fn new(url: String) -> Self {
44        Self {
45            url,
46            html_parse_success: false,
47            css_parse_success: false,
48            js_parse_success: false,
49            html_parse_time: Duration::ZERO,
50            css_parse_time: Duration::ZERO,
51            js_parse_time: Duration::ZERO,
52            elements_parsed: 0,
53            css_rules_parsed: 0,
54            js_statements_parsed: 0,
55            errors: Vec::new(),
56        }
57    }
58
59    /// Check if all parsing succeeded
60    pub fn all_success(&self) -> bool {
61        self.html_parse_success && self.css_parse_success && self.js_parse_success
62    }
63
64    /// Get total parse time
65    pub fn total_time(&self) -> Duration {
66        self.html_parse_time + self.css_parse_time + self.js_parse_time
67    }
68
69    /// Get success rate (0.0 to 1.0)
70    pub fn success_rate(&self) -> f64 {
71        let mut count = 0;
72        let mut success = 0;
73
74        count += 1;
75        if self.html_parse_success {
76            success += 1;
77        }
78        count += 1;
79        if self.css_parse_success {
80            success += 1;
81        }
82        count += 1;
83        if self.js_parse_success {
84            success += 1;
85        }
86
87        success as f64 / count as f64
88    }
89}

91/// Website tester for real-world validation
92pub struct WebsiteTester {
93    html_parser: HtmlParser,
94    css_parser: CssParser,
95    js_parser: JsParser,
96}
97
98impl WebsiteTester {
99    /// Create a new website tester
100    pub fn new() -> Self {
101        Self {
102            html_parser: HtmlParser::new(),
103            css_parser: CssParser::new(),
104            js_parser: JsParser::new(),
105        }
106    }
107
108    /// Test HTML content
109    pub fn test_html(&self, html: &str) -> (bool, Duration, usize) {
110        let start = Instant::now();
111        match self.html_parser.parse(html) {
112            Ok(dom) => {
113                let duration = start.elapsed();
114                // Count elements in DOM
115                let elements = Self::count_elements(&dom);
116                (true, duration, elements)
117            }
118            Err(_) => (false, start.elapsed(), 0),
119        }
120    }
121
122    /// Test CSS content
123    pub fn test_css(&self, css: &str) -> (bool, Duration, usize) {
124        let start = Instant::now();
125        match self.css_parser.parse(css) {
126            Ok(rules) => {
127                let duration = start.elapsed();
128                (true, duration, rules.len())
129            }
130            Err(_) => (false, start.elapsed(), 0),
131        }
132    }
133
134    /// Test JavaScript content
135    pub fn test_js(&self, js: &str) -> (bool, Duration, usize) {
136        let start = Instant::now();
137        match self.js_parser.parse(js) {
138            Ok(ast) => {
139                let duration = start.elapsed();
140                (true, duration, ast.statement_count)
141            }
142            Err(_) => (false, start.elapsed(), 0),
143        }
144    }
145
146    /// Test a complete website with HTML, CSS, and JS
147    pub fn test_website(&self, url: &str, html: &str, css: &str, js: &str) -> WebsiteTestResult {
148        let mut result = WebsiteTestResult::new(url.to_string());
149
150        // Test HTML
151        let (success, time, count) = self.test_html(html);
152        result.html_parse_success = success;
153        result.html_parse_time = time;
154        result.elements_parsed = count;
155        if !success {
156            result.errors.push("HTML parsing failed".to_string());
157        }
158
159        // Test CSS
160        let (success, time, count) = self.test_css(css);
161        result.css_parse_success = success;
162        result.css_parse_time = time;
163        result.css_rules_parsed = count;
164        if !success {
165            result.errors.push("CSS parsing failed".to_string());
166        }
167
168        // Test JS
169        let (success, time, count) = self.test_js(js);
170        result.js_parse_success = success;
171        result.js_parse_time = time;
172        result.js_statements_parsed = count;
173        if !success {
174            result.errors.push("JS parsing failed".to_string());
175        }
176
177        result
178    }
179
180    /// Count elements in DOM
181    fn count_elements(dom: &markup5ever_rcdom::RcDom) -> usize {
182        use markup5ever_rcdom::NodeData;
183        let mut count = 0;
184
185        fn walk_tree(node: &markup5ever_rcdom::Handle, count: &mut usize) {
186            if let NodeData::Element { .. } = node.data {
187                *count += 1;
188            }
189            for child in node.children.borrow().iter() {
190                walk_tree(child, count);
191            }
192        }
193
194        walk_tree(&dom.document, &mut count);
195        count
196    }
197}

205/// Test suite aggregator for multiple website tests
206pub struct WebsiteTestSuite {
207    results: Vec<WebsiteTestResult>,
208}
209
210impl WebsiteTestSuite {
211    /// Create a new test suite
212    pub fn new() -> Self {
213        Self {
214            results: Vec::new(),
215        }
216    }
217
218    /// Add a test result
219    pub fn add_result(&mut self, result: WebsiteTestResult) {
220        self.results.push(result);
221    }
222
223    /// Get all results
224    pub fn results(&self) -> &[WebsiteTestResult] {
225        &self.results
226    }
227
228    /// Get overall success rate
229    pub fn overall_success_rate(&self) -> f64 {
230        if self.results.is_empty() {
231            return 0.0;
232        }
233
234        let total_rate: f64 = self.results.iter().map(|r| r.success_rate()).sum();
235        total_rate / self.results.len() as f64
236    }
237
238    /// Get average parse time
239    pub fn average_parse_time(&self) -> Duration {
240        if self.results.is_empty() {
241            return Duration::ZERO;
242        }
243
244        let total: Duration = self.results.iter().map(|r| r.total_time()).sum();
245        total / self.results.len() as u32
246    }
247
248    /// Get total elements parsed
249    pub fn total_elements_parsed(&self) -> usize {
250        self.results.iter().map(|r| r.elements_parsed).sum()
251    }
252
253    /// Get total CSS rules parsed
254    pub fn total_css_rules_parsed(&self) -> usize {
255        self.results.iter().map(|r| r.css_rules_parsed).sum()
256    }
257
258    /// Get total JS statements parsed
259    pub fn total_js_statements_parsed(&self) -> usize {
260        self.results.iter().map(|r| r.js_statements_parsed).sum()
261    }
262
263    /// Generate a summary report
264    pub fn generate_report(&self) -> String {
265        let mut report = String::new();
266        report.push_str("=== Website Test Suite Report ===\n\n");
267
268        report.push_str(&format!("Total websites tested: {}\n", self.results.len()));
269        report.push_str(&format!(
270            "Overall success rate: {:.2}%\n",
271            self.overall_success_rate() * 100.0
272        ));
273        report.push_str(&format!(
274            "Average parse time: {:.2}ms\n",
275            self.average_parse_time().as_secs_f64() * 1000.0
276        ));
277        report.push_str(&format!(
278            "Total elements parsed: {}\n",
279            self.total_elements_parsed()
280        ));
281        report.push_str(&format!(
282            "Total CSS rules parsed: {}\n",
283            self.total_css_rules_parsed()
284        ));
285        report.push_str(&format!(
286            "Total JS statements parsed: {}\n\n",
287            self.total_js_statements_parsed()
288        ));
289
290        report.push_str("Individual Results:\n");
291        for result in &self.results {
292            report.push_str(&format!("\nURL: {}\n", result.url));
293            report.push_str(&format!(
294                "  Success Rate: {:.2}%\n",
295                result.success_rate() * 100.0
296            ));
297            report.push_str(&format!(
298                "  Total Time: {:.2}ms\n",
299                result.total_time().as_secs_f64() * 1000.0
300            ));
301            report.push_str(&format!("  Elements: {}\n", result.elements_parsed));
302            report.push_str(&format!("  CSS Rules: {}\n", result.css_rules_parsed));
303            report.push_str(&format!(
304                "  JS Statements: {}\n",
305                result.js_statements_parsed
306            ));
307            if !result.errors.is_empty() {
308                report.push_str(&format!("  Errors: {:?}\n", result.errors));
309            }
310        }
311
312        report
313    }
314}
