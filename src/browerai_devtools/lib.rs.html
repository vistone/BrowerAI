7/// DOM node information for inspection
8#[derive(Debug, Clone)]
9pub struct NodeInfo {
10    pub node_type: String,
11    pub tag_name: Option<String>,
12    pub attributes: HashMap<String, String>,
13    pub text_content: Option<String>,
14    pub child_count: usize,
15}

17/// DOM inspector for debugging
18pub struct DOMInspector {
19    dom: Option<RcDom>,
20}
21
22impl DOMInspector {
23    pub fn new() -> Self {
24        Self { dom: None }
25    }
26
27    pub fn set_dom(&mut self, dom: RcDom) {
28        self.dom = Some(dom);
29    }
30
31    fn get_node_info(&self, handle: &Handle) -> NodeInfo {
32        match &handle.data {
33            NodeData::Element { name, attrs, .. } => {
34                let mut attributes = HashMap::new();
35                for attr in attrs.borrow().iter() {
36                    attributes.insert(attr.name.local.to_string(), attr.value.to_string());
37                }
38                NodeInfo {
39                    node_type: "Element".to_string(),
40                    tag_name: Some(name.local.to_string()),
41                    attributes,
42                    text_content: None,
43                    child_count: handle.children.borrow().len(),
44                }
45            }
46            NodeData::Text { contents } => NodeInfo {
47                node_type: "Text".to_string(),
48                tag_name: None,
49                attributes: HashMap::new(),
50                text_content: Some(contents.borrow().to_string()),
51                child_count: 0,
52            },
53            NodeData::Document => NodeInfo {
54                node_type: "Document".to_string(),
55                tag_name: None,
56                attributes: HashMap::new(),
57                text_content: None,
58                child_count: handle.children.borrow().len(),
59            },
60            _ => NodeInfo {
61                node_type: "Other".to_string(),
62                tag_name: None,
63                attributes: HashMap::new(),
64                text_content: None,
65                child_count: 0,
66            },
67        }
68    }
69
70    pub fn tree_view(&self) -> Result<String> {
71        let dom = self
72            .dom
73            .as_ref()
74            .ok_or_else(|| anyhow::anyhow!("No DOM loaded"))?;
75        let mut output = String::new();
76        self.build_tree_view(&dom.document, 0, &mut output);
77        Ok(output)
78    }
79
80    fn build_tree_view(&self, handle: &Handle, depth: usize, output: &mut String) {
81        let indent = "  ".repeat(depth);
82        let info = self.get_node_info(handle);
83        match info.node_type.as_str() {
84            "Element" => {
85                output.push_str(&format!(
86                    "{}<{}> ({} children)\n",
87                    indent,
88                    info.tag_name.unwrap_or_default(),
89                    info.child_count
90                ));
91            }
92            "Text" => {
93                if let Some(text) = info.text_content {
94                    let trimmed = text.trim();
95                    if !trimmed.is_empty() {
96                        output.push_str(&format!("{}[Text: \"{}\"]\n", indent, trimmed));
97                    }
98                }
99            }
100            _ => {}
101        }
102        for child in handle.children.borrow().iter() {
103            self.build_tree_view(child, depth + 1, output);
104        }
105    }
106}
107
108impl Default for DOMInspector {
109    fn default() -> Self {
110        Self::new()
111    }
112}

114/// Network request record
115#[derive(Debug, Clone)]
116pub struct NetworkRequest {
117    pub url: String,
118    pub method: String,
119    pub status_code: Option<u16>,
120    pub size: usize,
121    pub duration: Duration,
122    pub timestamp: Instant,
123}
124
125/// Network monitor for tracking requests
126pub struct NetworkMonitor {
127    requests: Vec<NetworkRequest>,
128    total_bytes: usize,
129}
130
131impl NetworkMonitor {
132    pub fn new() -> Self {
133        Self {
134            requests: Vec::new(),
135            total_bytes: 0,
136        }
137    }
138
139    pub fn record_request(&mut self, request: NetworkRequest) {
140        self.total_bytes += request.size;
141        self.requests.push(request);
142    }
143
144    pub fn requests(&self) -> &[NetworkRequest] {
145        &self.requests
146    }
147
148    pub fn total_bytes(&self) -> usize {
149        self.total_bytes
150    }
151
152    pub fn request_count(&self) -> usize {
153        self.requests.len()
154    }
155
156    pub fn average_duration(&self) -> Duration {
157        if self.requests.is_empty() {
158            return Duration::from_secs(0);
159        }
160        let total: Duration = self.requests.iter().map(|r| r.duration).sum();
161        total / self.requests.len() as u32
162    }
163
164    pub fn clear(&mut self) {
165        self.requests.clear();
166        self.total_bytes = 0;
167    }
168
169    pub fn report(&self) -> String {
170        let mut output = String::new();
171        output.push_str("=== Network Monitor Report ===\n");
172        output.push_str(&format!("Total Requests: {}\n", self.request_count()));
173        output.push_str(&format!("Total Bytes: {} bytes\n", self.total_bytes()));
174        output.push_str(&format!(
175            "Average Duration: {:?}\n",
176            self.average_duration()
177        ));
178        output
179    }
180}
181
182impl Default for NetworkMonitor {
183    fn default() -> Self {
184        Self::new()
185    }
186}

188/// Performance metrics
189#[derive(Debug, Clone)]
190pub struct PerformanceMetrics {
191    pub parse_time: Duration,
192    pub layout_time: Duration,
193    pub paint_time: Duration,
194    pub total_time: Duration,
195}
196
197/// Performance profiler
198pub struct PerformanceProfiler {
199    metrics: Option<PerformanceMetrics>,
200}
201
202impl PerformanceProfiler {
203    pub fn new() -> Self {
204        Self { metrics: None }
205    }
206
207    pub fn record(&mut self, metrics: PerformanceMetrics) {
208        self.metrics = Some(metrics);
209    }
210
211    pub fn metrics(&self) -> Option<&PerformanceMetrics> {
212        self.metrics.as_ref()
213    }
214
215    pub fn report(&self) -> String {
216        let mut output = String::new();
217        output.push_str("=== Performance Profiler Report ===\n");
218        if let Some(metrics) = &self.metrics {
219            output.push_str(&format!("Parse Time: {:?}\n", metrics.parse_time));
220            output.push_str(&format!("Layout Time: {:?}\n", metrics.layout_time));
221            output.push_str(&format!("Paint Time: {:?}\n", metrics.paint_time));
222            output.push_str(&format!("Total Time: {:?}\n", metrics.total_time));
223        } else {
224            output.push_str("No metrics recorded yet.\n");
225        }
226        output
227    }
228}
229
230impl Default for PerformanceProfiler {
231    fn default() -> Self {
232        Self::new()
233    }
234}
