1//! Data Flow Analyzer - Advanced data flow and variable usage analysis
2//!
3//! This module provides sophisticated data flow analysis capabilities including:
4//! - Definition-use chain construction
5//! - Unused variable detection
6//! - Constant propagation analysis
7//! - Dead code identification
8//! - Reachability analysis
9
10use anyhow::{Context, Result};
11use std::collections::HashMap;
12
13use crate::extractor::ExtractedAst;
14use crate::types::{
15    DataFlowGraph, DataFlowNode, DataFlowNodeType, JsSemanticInfo, LocationInfo, ScopeTree,
16};
17
18/// Data flow analyzer for JavaScript code
19pub struct DataFlowAnalyzer {
20    /// Next node ID counter
21    next_node_id: usize,
22
23    /// Cache for def-use information
24    def_use_cache: HashMap<String, Vec<String>>,
25}
26
27impl DataFlowAnalyzer {
28    /// Create a new data flow analyzer
29    pub fn new() -> Self {
30        Self {
31            next_node_id: 0,
32            def_use_cache: HashMap::new(),
33        }
34    }
35
36    /// Analyze data flow in the given AST
37    pub fn analyze(&mut self, ast: &ExtractedAst, scope_tree: &ScopeTree) -> Result<DataFlowGraph> {
38        let mut graph = DataFlowGraph::new();
39
40        // Build def-use chains from semantic info
41        self.build_def_use_chains(&mut graph, &ast.semantic, scope_tree)
42            .context("Failed to build def-use chains")?;
43
44        // Detect unused variables
45        self.detect_unused_variables(&mut graph);
46
47        // Identify constant candidates
48        self.identify_constants(&mut graph);
49
50        Ok(graph)
51    }
52
53    /// Build definition-use chains from semantic information
54    fn build_def_use_chains(
55        &mut self,
56        graph: &mut DataFlowGraph,
57        semantic: &JsSemanticInfo,
58        scope_tree: &ScopeTree,
59    ) -> Result<()> {
60        // Process global variables
61        for global_var in &semantic.global_vars {
62            self.add_definition_node(
63                graph,
64                global_var,
65                &scope_tree.global_scope_id,
66                DataFlowNodeType::Definition,
67                None,
68                1, // Assume global defs at line 1
69            );
70        }
71
72        // Process functions
73        for func in &semantic.functions {
74            let func_scope_id = &func.id;
75
76            // Add function parameters as definitions
77            for param in &func.parameters {
78                self.add_definition_node(
79                    graph,
80                    &param.name,
81                    func_scope_id,
82                    DataFlowNodeType::Parameter,
83                    None,
84                    func.start_line,
85                );
86            }
87
88            // Add local variables as definitions
89            for local_var in &func.local_vars {
90                self.add_definition_node(
91                    graph,
92                    local_var,
93                    func_scope_id,
94                    DataFlowNodeType::Definition,
95                    None,
96                    func.start_line,
97                );
98            }
99
100            // Add captured variables as uses
101            for captured_var in &func.captured_vars {
102                self.add_use_node(graph, captured_var, func_scope_id, None, func.start_line);
103            }
104
105            // Add called functions as uses
106            for called_func in &func.called_functions {
107                self.add_use_node(graph, called_func, func_scope_id, None, func.start_line);
108            }
109        }
110
111        Ok(())
112    }
113
114    /// Add a definition node to the graph
115    fn add_definition_node(
116        &mut self,
117        graph: &mut DataFlowGraph,
118        var_name: &str,
119        scope_id: &str,
120        node_type: DataFlowNodeType,
121        location: Option<LocationInfo>,
122        line: usize,
123    ) {
124        let node_id = format!("dfn_{}", self.next_node_id);
125        self.next_node_id += 1;
126
127        let node = DataFlowNode {
128            id: node_id.clone(),
129            node_type,
130            variable: var_name.to_string(),
131            location,
132            scope_id: scope_id.to_string(),
133            line,
134        };
135
136        graph.nodes.insert(node_id, node);
137
138        // Update def-use cache
139        self.def_use_cache
140            .entry(var_name.to_string())
141            .or_default()
142            .push(var_name.to_string());
143    }
144
145    /// Add a use node to the graph
146    fn add_use_node(
147        &mut self,
148        graph: &mut DataFlowGraph,
149        var_name: &str,
150        scope_id: &str,
151        location: Option<LocationInfo>,
152        line: usize,
153    ) {
154        let node_id = format!("dfu_{}", self.next_node_id);
155        self.next_node_id += 1;
156
157        let node = DataFlowNode {
158            id: node_id.clone(),
159            node_type: DataFlowNodeType::Use,
160            variable: var_name.to_string(),
161            location,
162            scope_id: scope_id.to_string(),
163            line,
164        };
165
166        graph.nodes.insert(node_id, node);
167    }
168
169    /// Detect unused variables in the graph
170    fn detect_unused_variables(&self, graph: &mut DataFlowGraph) {
171        let mut used_vars: std::collections::HashSet<String> = std::collections::HashSet::new();
172
173        // Collect all used variables
174        for node in graph.nodes.values() {
175            if node.node_type == DataFlowNodeType::Use {
176                used_vars.insert(node.variable.clone());
177            }
178        }
179
180        // Find defined but not used variables
181        for node in graph.nodes.values() {
182            if (node.node_type == DataFlowNodeType::Definition
183                || node.node_type == DataFlowNodeType::Parameter)
184                && !used_vars.contains(&node.variable)
185                && !graph.unused_variables.contains(&node.variable)
186            {
187                graph.unused_variables.push(node.variable.clone());
188            }
189        }
190    }
191
192    /// Identify constant candidates (variables that are never reassigned)
193    fn identify_constants(&self, graph: &mut DataFlowGraph) {
194        for _var_name in &graph.unused_variables {
195            continue; // Skip unused vars
196        }
197
198        // Find variables with only one definition and no assignments
199        let mut var_def_counts: HashMap<String, usize> = HashMap::new();
200
201        for node in graph.nodes.values() {
202            if node.node_type == DataFlowNodeType::Definition {
203                *var_def_counts.entry(node.variable.clone()).or_insert(0) += 1;
204            }
205        }
206
207        // Variables with exactly one definition are candidates for const
208        for (var_name, count) in var_def_counts {
209            if count == 1 {
210                // Check if there are any assignments (reassignments)
211                let has_assignment = graph
212                    .nodes
213                    .values()
214                    .any(|n| n.variable == var_name && n.node_type == DataFlowNodeType::Assignment);
215
216                if !has_assignment && !graph.constant_candidates.contains(&var_name) {
217                    graph.constant_candidates.push(var_name);
218                }
219            }
220        }
221    }
222}
