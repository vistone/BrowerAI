1/// 语义分析器 - 从提取的AST生成高级语义信息
2///
3/// 这个模块对提取的AST信息进行深层分析，识别：
4/// - 函数之间的调用关系
5/// - 类的继承和实现关系
6/// - 事件处理器的绑定
7/// - 框架检测
8/// - 模块依赖
9use super::types::*;
10use anyhow::Result;
11use std::collections::{HashMap, HashSet};

13/// 语义分析器
14pub struct SemanticAnalyzer {
15    /// 已识别的框架列表
16    _frameworks: Vec<FrameworkSignature>,
17
18    /// 事件处理器模式
19    _event_patterns: Vec<EventPattern>,
20}

22/// 框架签名 - 用于识别特定框架
23struct FrameworkSignature {
24    _name: &'static str,
25    _keywords: Vec<&'static str>,
26    _patterns: Vec<&'static str>,
27}

29/// 事件模式 - 用于识别事件处理器
30struct EventPattern {
31    _event_type: &'static str,

33}

50    /// 分析已提取的语义信息
51    pub fn analyze(&self, semantic: &mut JsSemanticInfo) -> Result<AnalysisResult> {
52        let start = std::time::Instant::now();
53
54        // 1. 检测框架
55        let detected_frameworks = self.detect_frameworks(semantic);
56        semantic.detected_frameworks = detected_frameworks.clone();
57
58        // 2. 检测特殊特性
59        let special_features = self.detect_special_features(semantic);
60        semantic.special_features = special_features.clone();
61
62        // 3. 分析函数属性
63        self.analyze_functions(semantic);
64
65        // 4. 检测事件处理器
66        let event_handlers_found = semantic.event_handlers.len();
67
68        Ok(AnalysisResult {
69            detected_frameworks,
70            special_features,
71            event_handlers_found,
72            analysis_duration_ms: start.elapsed().as_millis() as u64,
73        })
74    }

76    /// Phase 2 增强：使用 EnhancedAst 进行更精细的分析
77    ///
78    /// 利用位置信息、JSX 和 TypeScript 数据改进分析精度
79    pub fn analyze_with_enhanced_ast(
80        &self,
81        semantic: &mut JsSemanticInfo,
82        enhanced: &crate::swc_extractor::EnhancedAst,
83    ) -> Result<AnalysisResult> {
84        // 首先执行标准分析
85        let mut result = self.analyze(semantic)?;
86
87        // Phase 2 增强：利用 JSX 和 TypeScript 信息
88
89        // 1. 改进 React 检测（如果有 JSX）
90        if enhanced.has_jsx && !result.detected_frameworks.contains(&"React".to_string()) {
91            // JSX 的存在强烈表明使用 React 或类似框架
92            result.detected_frameworks.push("React".to_string());
93            semantic.detected_frameworks.push("React".to_string());
94        }
95
96        // 2. 改进 TypeScript 检测
97        if enhanced.has_typescript {
98            result
99                .special_features
100                .push("typescript_support".to_string());
101            semantic
102                .special_features
103                .push("typescript_support".to_string());
104        }
105
106        // 3. 利用位置信息改进精度
107        // 将 JSX 元素信息关联到相应的组件函数
108        for jsx_elem in &enhanced.jsx_elements {
109            if jsx_elem.is_component {
110                // 记录 JSX 组件使用
111                log::debug!(
112                    "Found JSX component: {} at line {}",
113                    jsx_elem.name,
114                    jsx_elem.location.line
115                );
116            }
117        }
118
119        // 4. 利用 TypeScript 类型信息
120        for ts_type in &enhanced.typescript_types {
121            log::debug!(
122                "Found TypeScript {}: {} at line {}",
123                ts_type.kind,
124                ts_type.name,
125                ts_type.location.line
126            );
127
128            // Interface 的存在表明可能在使用 TypeScript
129            if ts_type.kind == "interface" {
130                result
131                    .special_features
132                    .push("typescript_interfaces".to_string());
133            }
134        }
135
136        // 5. 去重和排序
137        result.detected_frameworks.sort();
138        result.detected_frameworks.dedup();
139        result.special_features.sort();
140        result.special_features.dedup();
141
142        Ok(result)
143    }

145    /// 分析函数细节（如调用关系、复杂度等）
146    fn analyze_functions(&self, semantic: &mut JsSemanticInfo) {
147        for func in &mut semantic.functions {
148            // 基于名称推断可能的调用目标
149            let name = func.name.as_deref().unwrap_or("");
150
151            // 识别常见的回调函数
152            if name.starts_with("on") || name.contains("callback") {
153                semantic.event_handlers.push(JsEventHandler {
154                    id: func.id.clone(),
155                    event_type: self.infer_event_type(name),
156                    handler_function_id: func.id.clone(),
157                    target_selector: None,
158                    is_delegated: false,
159                    binding_method: EventBindingMethod::DirectProperty,
160                });
161            }
162        }
163    }

165    /// 推断事件类型
166    fn infer_event_type(&self, func_name: &str) -> String {
167        let lower = func_name.to_lowercase();
168
169        // 检查 camelCase 命名模式：onXxx
170        if lower.starts_with("on") && lower.len() > 2 {
171            let event = &lower[2..]; // 去掉 "on" 前缀
172            return event.to_string();
173        }
174
175        // 检查 handleXxx 模式
176        if lower.starts_with("handle") && lower.len() > 6 {
177            let event = &lower[6..]; // 去掉 "handle" 前缀
178            return event.to_string();
179        }
180
181        "unknown".to_string()
182    }

184    /// 检测框架
185    fn detect_frameworks(&self, semantic: &JsSemanticInfo) -> Vec<String> {
186        let mut detected = HashSet::new();
187
188        // 检查全局变量
189        for var in &semantic.global_vars {
190            match var.as_str() {
191                "React" | "ReactDOM" => detected.insert("React".to_string()),
192                "Vue" => detected.insert("Vue".to_string()),
193                "angular" | "ng" => detected.insert("Angular".to_string()),
194                "$" | "jQuery" => detected.insert("jQuery".to_string()),
195                "_" => detected.insert("Lodash".to_string()),
196                _ => continue,
197            };
198        }
199
200        // 检查函数模式
201        for func in &semantic.functions {
202            if let Some(name) = &func.name {
203                // React hooks
204                if name.starts_with("use") && name.len() > 3 {
205                    detected.insert("React".to_string());
206                }
207                // Vue lifecycle
208                if matches!(
209                    name.as_str(),
210                    "beforeCreate"
211                        | "created"
212                        | "beforeMount"
213                        | "mounted"
214                        | "beforeUpdate"
215                        | "updated"
216                        | "beforeDestroy"
217                        | "destroyed"
218                ) {
219                    detected.insert("Vue".to_string());
220                }
221            }
222        }
223
224        detected.into_iter().collect()
225    }

227    /// 检测特殊特性
228    fn detect_special_features(&self, semantic: &JsSemanticInfo) -> Vec<String> {
229        let mut features = vec![];
230
231        if semantic.uses_eval {
232            features.push("dynamic_eval".to_string());
233        }
234
235        if semantic.uses_dynamic_require {
236            features.push("dynamic_require".to_string());
237        }
238
239        // 检查异步与生成器
240        let has_async = semantic.functions.iter().any(|f| f.is_async);
241        let has_generator = semantic.functions.iter().any(|f| f.is_generator);
242
243        if has_async {
244            features.push("async_functions".to_string());
245        }
246        if has_generator {
247            features.push("generators".to_string());
248        }
249
250        // 检查高复杂度
251        if semantic
252            .functions
253            .iter()
254            .any(|f| f.cyclomatic_complexity > 10)
255        {
256            features.push("high_complexity".to_string());
257        }
258
259        // 检查大量函数
260        if semantic.functions.len() > 100 {
261            features.push("many_functions".to_string());
262        }
263
264        features.sort();
265        features.dedup();
266        features
267    }

327/// 函数调用分析结果
328#[derive(Debug, Clone)]
329pub struct FunctionCallAnalysis {
330    /// 函数ID到其调用的函数列表的映射
331    pub call_graph: HashMap<String, Vec<String>>,
332
333    /// 循环调用检测
334    pub circular_calls: Vec<Vec<String>>,
335
336    /// 入口点函数
337    pub entry_points: Vec<String>,
338}
339
340impl FunctionCallAnalysis {
341    /// 检测循环调用
342    pub fn detect_cycles(&self) -> Vec<Vec<String>> {
343        let mut cycles = vec![];
344        let mut visited = HashSet::new();
345        let mut rec_stack = HashSet::new();
346
347        for start_node in self.call_graph.keys() {
348            if !visited.contains(start_node) {
349                Self::dfs(
350                    start_node,
351                    &self.call_graph,
352                    &mut visited,
353                    &mut rec_stack,
354                    &mut vec![],
355                    &mut cycles,
356                );
357            }
358        }
359
360        cycles
361    }
362
363    /// DFS算法用于循环检测
364    fn dfs(
365        node: &str,
366        graph: &HashMap<String, Vec<String>>,
367        visited: &mut HashSet<String>,
368        rec_stack: &mut HashSet<String>,
369        path: &mut Vec<String>,
370        cycles: &mut Vec<Vec<String>>,
371    ) {
372        visited.insert(node.to_string());
373        rec_stack.insert(node.to_string());
374        path.push(node.to_string());
375
376        if let Some(neighbors) = graph.get(node) {
377            for neighbor in neighbors {
378                if !visited.contains(neighbor) {
379                    Self::dfs(neighbor, graph, visited, rec_stack, path, cycles);
380                } else if rec_stack.contains(neighbor) {
381                    // 找到循环
382                    if let Some(idx) = path.iter().position(|n| n == neighbor) {
383                        let cycle = path[idx..].to_vec();
384                        cycles.push(cycle);
385                    }
386                }
387            }
388        }
389
390        path.pop();
391        rec_stack.remove(node);
392    }
