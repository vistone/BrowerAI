1//! Control Flow Analyzer - Advanced control flow graph and reachability analysis
2//!
3//! This module provides sophisticated control flow analysis including:
4//! - Control Flow Graph (CFG) construction
5//! - Reachability analysis
6//! - Loop detection and analysis
7//! - Dead code identification
8//! - Branch coverage analysis

18/// Control flow analyzer for JavaScript code

30    /// Analyze control flow in the given AST
31    pub fn analyze(&mut self, ast: &ExtractedAst) -> Result<ControlFlowGraph> {
32        let mut graph = ControlFlowGraph::new();
33
34        // Analyze each function
35        for func in &ast.semantic.functions {
36            self.analyze_function(&mut graph, func)?;
37        }
38
39        // Analyze each class method
40        for class in &ast.semantic.classes {
41            for _method in &class.methods {
42                // For now, treat methods as simple statements
43            }
44        }
45
46        // Perform reachability analysis
47        self.compute_reachability(&mut graph)?;
48
49        // Detect loops
50        self.detect_loops(&mut graph)?;
51
52        Ok(graph)
53    }

55    /// Analyze control flow in a function
56    fn analyze_function(
57        &mut self,
58        graph: &mut ControlFlowGraph,
59        func: &JsFunctionInfo,
60    ) -> Result<()> {
61        // Create entry node
62        let entry_id = self.create_node_id();
63        let entry_node = CFGNode {
64            id: entry_id.clone(),
65            node_type: CFGNodeType::Entry,
66            statement: None,
67            line: func.start_line,
68            column: 0,
69            is_reachable: true,
70        };
71        graph.entry = Some(entry_id.clone());
72        graph.nodes.push(entry_node);
73
74        // Create exit node
75        let exit_id = self.create_node_id();
76        let exit_node = CFGNode {
77            id: exit_id.clone(),
78            node_type: CFGNodeType::Exit,
79            statement: None,
80            line: func.end_line,
81            column: 0,
82            is_reachable: false, // Will be marked reachable if needed
83        };
84        graph.exit = Some(exit_id.clone());
85        graph.nodes.push(exit_node);
86
87        // Create nodes for function body statements
88        let mut current_node = entry_id.clone();
89
90        // Add a statement node for function parameters/setup
91        let stmt_id = self.create_node_id();
92        let stmt_node = CFGNode {
93            id: stmt_id.clone(),
94            node_type: CFGNodeType::Statement,
95            statement: Some(format!(
96                "function {}(...)",
97                func.name.as_deref().unwrap_or("anonymous")
98            )),
99            line: func.start_line,
100            column: 0,
101            is_reachable: true,
102        };
103        graph.nodes.push(stmt_node);
104
105        // Create edge from entry to first statement
106        graph.edges.push(CFGEdge {
107            from: current_node.clone(),
108            to: stmt_id.clone(),
109            edge_type: EdgeType::Unconditional,
110        });
111        current_node = stmt_id;
112
113        // Create nodes for local variables
114        for local_var in &func.local_vars {
115            let var_id = self.create_node_id();
116            let var_node = CFGNode {
117                id: var_id.clone(),
118                node_type: CFGNodeType::Statement,
119                statement: Some(format!("let {} = ...;", local_var)),
120                line: func.start_line + 1,
121                column: 0,
122                is_reachable: true,
123            };
124            graph.nodes.push(var_node);
125
126            // Create edge from previous node
127            graph.edges.push(CFGEdge {
128                from: current_node.clone(),
129                to: var_id.clone(),
130                edge_type: EdgeType::Unconditional,
131            });
132            current_node = var_id;
133        }
134
135        // Create nodes for called functions
136        for called_func in &func.called_functions {
137            let call_id = self.create_node_id();
138            let call_node = CFGNode {
139                id: call_id.clone(),
140                node_type: CFGNodeType::Statement,
141                statement: Some(format!("{}(...);", called_func)),
142                line: func.start_line + 2,
143                column: 0,
144                is_reachable: true,
145            };
146            graph.nodes.push(call_node);
147
148            // Create edge from previous node
149            graph.edges.push(CFGEdge {
150                from: current_node.clone(),
151                to: call_id.clone(),
152                edge_type: EdgeType::Unconditional,
153            });
154            current_node = call_id;
155        }
156
157        // Connect final statement to exit
158        graph.edges.push(CFGEdge {
159            from: current_node,
160            to: exit_id,
161            edge_type: EdgeType::Unconditional,
162        });
163
164        Ok(())
165    }

167    /// Compute reachability for all nodes
168    fn compute_reachability(&self, graph: &mut ControlFlowGraph) -> Result<()> {
169        let mut reachable = HashSet::new();
170        let mut queue = VecDeque::new();
171
172        // Start from entry node
173        if let Some(entry_id) = &graph.entry {
174            queue.push_back(entry_id.clone());
175            reachable.insert(entry_id.clone());
176        }
177
178        // BFS to mark reachable nodes
179        while let Some(node_id) = queue.pop_front() {
180            // Find all successors
181            let successor_ids: Vec<_> = graph
182                .edges
183                .iter()
184                .filter(|e| e.from == node_id)
185                .map(|e| e.to.clone())
186                .collect();
187
188            for succ_id in successor_ids {
189                if !reachable.contains(&succ_id) {
190                    reachable.insert(succ_id.clone());
191                    queue.push_back(succ_id);
192                }
193            }
194        }
195
196        // Mark reachable nodes
197        for node in &mut graph.nodes {
198            node.is_reachable = reachable.contains(&node.id);
199        }
200
201        // Mark unreachable nodes
202        graph.unreachable_nodes = graph
203            .nodes
204            .iter()
205            .filter(|n| !n.is_reachable && n.node_type != CFGNodeType::Entry)
206            .map(|n| n.id.clone())
207            .collect();
208
209        Ok(())
210    }

212    /// Detect loops in the control flow graph
213    fn detect_loops(&mut self, graph: &mut ControlFlowGraph) -> Result<()> {
214        // Build adjacency list
215        let mut adj: std::collections::HashMap<String, Vec<String>> =
216            std::collections::HashMap::new();
217
218        for node in &graph.nodes {
219            adj.insert(node.id.clone(), Vec::new());
220        }
221
222        for edge in &graph.edges {
223            adj.entry(edge.from.clone())
224                .or_default()
225                .push(edge.to.clone());
226        }
227
228        // Detect back edges (simple loop detection)
229        let mut visited = HashSet::new();
230        let mut rec_stack = HashSet::new();
231        let nodes_copy: Vec<_> = graph.nodes.iter().map(|n| n.id.clone()).collect();
232
233        for node_id in nodes_copy {
234            if !visited.contains(&node_id) {
235                self.detect_loops_dfs(&node_id, &adj, &mut visited, &mut rec_stack, graph)?;
236            }
237        }
238
239        Ok(())
240    }

242    /// DFS for loop detection
243    fn detect_loops_dfs(
244        &self,
245        node_id: &str,
246        adj: &std::collections::HashMap<String, Vec<String>>,
247        visited: &mut HashSet<String>,
248        rec_stack: &mut HashSet<String>,
249        graph: &mut ControlFlowGraph,
250    ) -> Result<()> {
251        visited.insert(node_id.to_string());
252        rec_stack.insert(node_id.to_string());
253
254        if let Some(neighbors) = adj.get(node_id) {
255            let neighbors_copy = neighbors.clone();
256            for neighbor in neighbors_copy {
257                if rec_stack.contains(&neighbor) {
258                    // Back edge found - this is a loop
259                    let loop_info = LoopInfo {
260                        header: neighbor.clone(),
261                        latch: Some(node_id.to_string()),
262                        body_nodes: vec![],
263                        loop_type: LoopType::Other,
264                    };
265                    graph.loops.push(loop_info);
266
267                    // Mark edge as back edge
268                    if let Some(edge) = graph
269                        .edges
270                        .iter_mut()
271                        .find(|e| e.from == node_id && e.to == neighbor)
272                    {
273                        edge.edge_type = EdgeType::BackEdge;
274                    }
275                } else if !visited.contains(&neighbor) {
276                    self.detect_loops_dfs(&neighbor, adj, visited, rec_stack, graph)?;
277                }
278            }
279        }
280
281        rec_stack.remove(node_id);
282        Ok(())
283    }

285    /// Create a unique node ID
286    fn create_node_id(&mut self) -> String {
287        let id = format!("cfg_node_{}", self.next_node_id);
288        self.next_node_id += 1;
289        id
290    }
