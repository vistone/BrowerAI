1//! Advanced Loop Analysis Module
2//!
3//! This module provides comprehensive loop analysis capabilities including:
4//! - Induction variable detection (归纳变量检测)
5//! - Loop invariant identification (循环不变量识别)
6//! - Termination condition analysis (终止条件分析)
7//! - Iteration count estimation (迭代次数估计)
8//! - Nested loop analysis (嵌套循环分析)
9
10use super::extractor::ExtractedAst;
11use super::types::{ControlFlowGraph, DataFlowGraph, LoopInfo, ScopeTree};
12use anyhow::Result;
13
14/// Loop analysis result containing detailed information about loops
15#[derive(Debug, Clone)]
16pub struct LoopAnalysis {
17    pub loop_id: String,
18    pub loop_type: LoopType,
19    pub induction_variables: Vec<InductionVariable>,
20    pub invariants: Vec<String>,
21    pub termination_conditions: Vec<String>,
22    pub iteration_count_estimate: IterationEstimate,
23    pub nested_loops: Vec<String>,
24    pub complexity_score: u32,
25    pub is_potentially_infinite: bool,
26    pub early_exits: Vec<String>,
27}
28
29#[derive(Debug, Clone, PartialEq, Eq)]
30pub enum LoopType {
31    For,
32    While,
33    DoWhile,
34    IteratorLoop,
35}
36
37#[derive(Debug, Clone)]
38pub struct InductionVariable {
39    pub name: String,
40    pub update_pattern: UpdatePattern,
41    pub initial_value: Option<String>,
42    pub step: Option<String>,
43}
44
45#[derive(Debug, Clone, PartialEq, Eq)]
46pub enum UpdatePattern {
47    Increment,
48    Decrement,
49    Multiply,
50    Divide,
51    Complex,
52}
53
54#[derive(Debug, Clone, PartialEq, Eq)]
55pub enum IterationEstimate {
56    Fixed(usize),
57    Bounded(usize, usize),
58    Unbounded,
59    RuntimeDependent,
60}
61
62pub struct LoopAnalyzer {
63    loop_counter: usize,
64}
65
66impl LoopAnalyzer {
67    pub fn new() -> Self {
68        Self { loop_counter: 0 }
69    }
70
71    pub fn analyze(
72        &mut self,
73        _ast: &ExtractedAst,
74        _scope_tree: &ScopeTree,
75        _data_flow: &DataFlowGraph,
76        control_flow: &ControlFlowGraph,
77    ) -> Result<Vec<LoopAnalysis>> {
78        let mut analyses = Vec::new();
79        for loop_info in &control_flow.loops {
80            let analysis = self.analyze_single_loop(loop_info, control_flow)?;
81            analyses.push(analysis);
82        }
83        Ok(analyses)
84    }
85
86    fn analyze_single_loop(
87        &mut self,
88        loop_info: &LoopInfo,
89        cfg: &ControlFlowGraph,
90    ) -> Result<LoopAnalysis> {
91        let loop_id = self.generate_loop_id();
92        let induction_variables = self.detect_induction_variables(loop_info, cfg);
93        let invariants = self.identify_invariants(loop_info, cfg);
94        let termination_conditions = self.extract_termination_conditions(loop_info);
95        let iteration_count_estimate = self.estimate_iterations(loop_info, &induction_variables);
96        let nested_loops = self.find_nested_loops(loop_info, cfg);
97        let complexity_score =
98            self.calculate_complexity(loop_info, &induction_variables, &nested_loops);
99        let is_potentially_infinite = self.check_infinite_loop(loop_info, &termination_conditions);
100        let early_exits = self.detect_early_exits(loop_info, cfg);
101
102        Ok(LoopAnalysis {
103            loop_id,
104            loop_type: self.determine_loop_type(loop_info),
105            induction_variables,
106            invariants,
107            termination_conditions,
108            iteration_count_estimate,
109            nested_loops,
110            complexity_score,
111            is_potentially_infinite,
112            early_exits,
113        })
114    }
115
116    fn detect_induction_variables(
117        &self,
118        _loop_info: &LoopInfo,
119        _cfg: &ControlFlowGraph,
120    ) -> Vec<InductionVariable> {
121        Vec::new()
122    }
123
124    fn identify_invariants(&self, _loop_info: &LoopInfo, _cfg: &ControlFlowGraph) -> Vec<String> {
125        Vec::new()
126    }
127
128    fn extract_termination_conditions(&self, _loop_info: &LoopInfo) -> Vec<String> {
129        Vec::new()
130    }
131
132    fn estimate_iterations(
133        &self,
134        _loop_info: &LoopInfo,
135        induction_variables: &[InductionVariable],
136    ) -> IterationEstimate {
137        if induction_variables.is_empty() {
138            IterationEstimate::RuntimeDependent
139        } else {
140            IterationEstimate::Unbounded
141        }
142    }
143
144    fn find_nested_loops(&self, loop_info: &LoopInfo, cfg: &ControlFlowGraph) -> Vec<String> {
145        cfg.loops
146            .iter()
147            .filter(|l| l.header != loop_info.header && loop_info.body_nodes.contains(&l.header))
148            .map(|l| l.header.clone())
149            .collect()
150    }
151
152    fn calculate_complexity(
153        &self,
154        loop_info: &LoopInfo,
155        induction_variables: &[InductionVariable],
156        nested_loops: &[String],
157    ) -> u32 {
158        let mut score = 10;
159        score += (loop_info.body_nodes.len() as u32).min(30);
160        score += (induction_variables.len() as u32 * 5).min(20);
161        score += (nested_loops.len() as u32 * 15).min(40);
162        score.min(100)
163    }
164
165    fn check_infinite_loop(
166        &self,
167        _loop_info: &LoopInfo,
168        termination_conditions: &[String],
169    ) -> bool {
170        termination_conditions.is_empty()
171    }
172
173    fn detect_early_exits(&self, loop_info: &LoopInfo, _cfg: &ControlFlowGraph) -> Vec<String> {
174        if loop_info.latch.is_none() {
175            vec!["implicit_exit".to_string()]
176        } else {
177            Vec::new()
178        }
179    }
180
181    fn determine_loop_type(&self, loop_info: &LoopInfo) -> LoopType {
182        match loop_info.loop_type {
183            super::types::LoopType::For => LoopType::For,
184            super::types::LoopType::While => LoopType::While,
185            super::types::LoopType::DoWhile => LoopType::DoWhile,
186            super::types::LoopType::ForIn => LoopType::IteratorLoop,
187            super::types::LoopType::Other => LoopType::For,
188        }
189    }
190
191    fn generate_loop_id(&mut self) -> String {
192        let id = format!("loop_{}", self.loop_counter);
193        self.loop_counter += 1;
194        id
195    }
196}

224        let code = "function test() { for (let i = 0; i < 10; i++) { console.log(i); } }";
225        let ast = extract_ast(code);

227        let data_flow = DataFlowAnalyzer::new().analyze(&ast, &scope_tree).unwrap();
228        let control_flow = ControlFlowAnalyzer::new().analyze(&ast).unwrap();

330    fn test_complexity_calculation() {
331        let analyzer = LoopAnalyzer::new();
332        let simple_loop = LoopInfo {
333            header: "h1".to_string(),
334            latch: Some("latch1".to_string()),

339        assert!(simple_complexity < 30);
340
341        let nested_loops = vec!["inner1".to_string(), "inner2".to_string()];

343        assert!(complex_complexity > simple_complexity);
