92/// Swc 完整 AST 提取器
93///
94/// 设计特点：
95/// - 兼容 Phase 1 的输出格式
96/// - 使用 Swc 进行完整 AST 解析
97/// - 支持 JSX 和 TypeScript
98/// - 获取精确的位置信息
99///
100/// # 示例
101///
102/// ```ignore
103/// let extractor = SwcAstExtractor::new();
104/// let enhanced = extractor.extract_from_source(code)?;
105///
106/// // 访问位置信息
107/// if let Some(loc) = enhanced.locations.get("myFunction") {
108///     println!("Function at line {}, column {}", loc.line, loc.column);
109/// }
110///
111/// // 访问 JSX 信息
112/// for jsx in &enhanced.jsx_elements {
113///     println!("JSX element: <{} />", jsx.name);
114/// }
115/// ```
116#[derive(Default)]
117pub struct SwcAstExtractor {
118    // 预留位置：用于 Swc 配置和状态
119    _phantom: std::marker::PhantomData<()>,
120}

127    /// 从源代码提取增强的 AST 信息
128    ///
129    /// # 参数
130    /// - `source` - JavaScript/TypeScript 源代码
131    ///
132    /// # 返回
133    /// 增强的 AST 数据，包括位置信息、JSX 和 TypeScript 支持
134    ///
135    /// # 错误
136    /// 如果解析失败，返回包含详细信息的错误
137    pub fn extract_from_source(&self, source: &str) -> Result<EnhancedAst> {
138        // 第一步：先使用 Phase 1 的提取器获取基础数据
139        let mut extractor = AstExtractor::new();
140        let base = extractor.extract_from_source(source)?;
141
142        // 第二步：检测 TypeScript（根据文件内容）
143        let has_typescript = self.detect_typescript(source);
144
145        // 第三步：使用 Swc 进行完整解析
146        let mut swc_result = self.parse_with_swc(source, has_typescript)?;
147
148        // 第四步：提取精确的位置信息
149        if swc_result.locations.is_empty() {
150            swc_result.locations = self.extract_locations(source);
151        }
152
153        let enhanced = EnhancedAst {
154            metadata: base.metadata,
155            semantic: base.semantic,
156            locations: swc_result.locations,
157            jsx_elements: swc_result.jsx_elements,
158            typescript_types: swc_result.typescript_types,
159            has_jsx: swc_result.has_jsx,
160            has_typescript,
161            is_modern_js: true,
162        };
163
164        log::debug!(
165            "Enhanced AST: JSX={}, TS={}, locations={}, jsx_count={}, ts_count={}",
166            enhanced.has_jsx,
167            enhanced.has_typescript,
168            enhanced.locations.len(),
169            enhanced.jsx_elements.len(),
170            enhanced.typescript_types.len()
171        );
172
173        Ok(enhanced)
174    }

176    /// 使用 Swc 进行完整的 AST 解析
177    ///
178    /// 返回包含位置信息、JSX 和 TypeScript 数据的结果
179    ///
180    /// 注意：当前使用启发式实现。完整的 Swc AST 遍历将在 Phase 2 Day 3 实现。
181    fn parse_with_swc(&self, source: &str, _is_typescript: bool) -> Result<SwcParseResult> {
182        // Phase 2 Day 2-3：基础启发式实现
183        // Phase 2 Day 3：完整的 Swc AST 遍历和位置信息提取
184        //
185        // 当前使用启发式方法：
186        // 1. 检测 JSX - 通过 <Component /> 模式
187        // 2. 检测 TypeScript - 通过类型注解
188        // 3. 提取类型定义 - 通过正则表达式
189        // 4. 提取元素位置 - 基于文本搜索
190
191        Ok(SwcParseResult {
192            has_jsx: self.detect_jsx(source),
193            jsx_elements: self.extract_jsx_elements_heuristic(source),
194            typescript_types: self.extract_typescript_heuristic(source),
195            locations: HashMap::new(), // 将在 Day 3 实现精确位置信息
196        })
197    }

199    /// 使用启发式方法检测 TypeScript
200    fn detect_typescript(&self, source: &str) -> bool {
201        // 检查常见的 TypeScript 特征
202        source.contains(": string")
203            || source.contains(": number")
204            || source.contains(": boolean")
205            || source.contains(": any")
206            || source.contains("interface ")
207            || source.contains("type ")
208            || source.contains("<T>")
209            || source.contains("as const")
210            || source.contains("enum ")
211    }

213    /// 使用启发式方法检测 JSX
214    fn detect_jsx(&self, source: &str) -> bool {
215        // 检查 JSX 特征：<Component ... /> 或 <div ...>
216        let has_jsx_like = source.contains("</") || (source.contains("<") && source.contains("/>"));
217        let has_react_imports = source.contains("import React") || source.contains("from 'react'");
218
219        has_jsx_like || has_react_imports
220    }

222    /// 使用启发式方法提取 JSX 元素信息
223    fn extract_jsx_elements_heuristic(&self, source: &str) -> Vec<JsxElementInfo> {
224        let mut elements = Vec::new();
225
226        // 简单的正则表达式匹配：<ComponentName ... >
227        if let Ok(re) = regex::Regex::new(r#"<([A-Za-z][A-Za-z0-9]*)\s*([^>]*)/?>"#) {
228            for cap in re.captures_iter(source) {
229                if cap.len() >= 2 {
230                    let name = cap
231                        .get(1)
232                        .map(|m| m.as_str())
233                        .unwrap_or("Unknown")
234                        .to_string();
235                    let is_component = name
236                        .chars()
237                        .next()
238                        .map(|c| c.is_uppercase())
239                        .unwrap_or(false);
240                    let attrs_str = cap.get(2).map(|m| m.as_str()).unwrap_or("");
241
242                    let attributes = self.parse_jsx_attributes(attrs_str);
243
244                    // 计算精确的位置信息
245                    let start_pos = cap.get(0).unwrap().start();
246                    let end_pos = cap.get(0).unwrap().end();
247                    let location = self.compute_location_from_offsets(source, start_pos, end_pos);
248
249                    elements.push(JsxElementInfo {
250                        name,
251                        is_component,
252                        attributes,
253                        location,
254                    });
255                }
256            }
257        }
258
259        elements
260    }

262    /// 使用启发式方法提取 TypeScript 类型信息
263    fn extract_typescript_heuristic(&self, source: &str) -> Vec<TypeScriptInfo> {
264        let mut types = Vec::new();
265
266        // 提取 interface 定义
267        if let Ok(re) =
268            regex::Regex::new(r"interface\s+(\w+)\s*(?:<[^>]+>)?\s*(?:extends\s+([^{]+))?\s*\{")
269        {
270            for cap in re.captures_iter(source) {
271                if let Some(name) = cap.get(1) {
272                    let extends = cap
273                        .get(2)
274                        .map(|m| {
275                            m.as_str()
276                                .split(',')
277                                .map(|s| s.trim().to_string())
278                                .collect()
279                        })
280                        .unwrap_or_default();
281
282                    types.push(TypeScriptInfo {
283                        name: name.as_str().to_string(),
284                        kind: "interface".to_string(),
285                        extends,
286                        location: LocationInfo::new(1, 0, 0, 10),
287                    });
288                }
289            }
290        }
291
292        // 提取 type 定义
293        if let Ok(re) = regex::Regex::new(r"type\s+(\w+)\s*(?:<[^>]+>)?\s*=") {
294            for cap in re.captures_iter(source) {
295                if let Some(name) = cap.get(1) {
296                    types.push(TypeScriptInfo {
297                        name: name.as_str().to_string(),
298                        kind: "type".to_string(),
299                        extends: Vec::new(),
300                        location: LocationInfo::new(1, 0, 0, 10),
301                    });
302                }
303            }
304        }
305
306        types
307    }

309    /// 解析 JSX 属性字符串
310    fn parse_jsx_attributes(&self, attrs_str: &str) -> Vec<JsxAttribute> {
311        let mut attributes = Vec::new();
312
313        // 简单的属性解析：key="value" 或 key
314        if let Ok(re) = regex::Regex::new(r#"(\w+)(?:=(?:"([^"]*)"|'([^']*)'|(\S+)))?"#) {
315            for cap in re.captures_iter(attrs_str) {
316                if let Some(key) = cap.get(1) {
317                    let value = cap
318                        .get(2)
319                        .or_else(|| cap.get(3))
320                        .or_else(|| cap.get(4))
321                        .map(|m| m.as_str().to_string());
322
323                    attributes.push(JsxAttribute {
324                        name: key.as_str().to_string(),
325                        value,
326                    });
327                }
328            }
329        }
330
331        attributes
332    }

334    /// 从字符偏移计算位置信息 (line, column)
335    fn compute_location_from_offsets(
336        &self,
337        source: &str,
338        start: usize,
339        end: usize,
340    ) -> LocationInfo {
341        let before = &source[..start.min(source.len())];
342        let line = before.lines().count();
343        let last_line = before.lines().last().unwrap_or("");
344        let column = last_line.len();
345
346        LocationInfo::new(line, column, start, end)
347    }

349    /// 提取所有函数和类的位置信息
350    fn extract_locations(&self, source: &str) -> HashMap<String, LocationInfo> {
351        let mut locations = HashMap::new();
352
353        // 提取函数声明：function name() 或 const/let name = ()=>
354        if let Ok(re) = regex::Regex::new(r"(?:function|const|let)\s+(\w+)\s*(?:\(|=)") {
355            for cap in re.captures_iter(source) {
356                if let Some(name) = cap.get(1) {
357                    let name_str = name.as_str();
358                    if let Some(pos) = source.find(name_str) {
359                        let location =
360                            self.compute_location_from_offsets(source, pos, pos + name_str.len());
361                        locations.insert(name_str.to_string(), location);
362                    }
363                }
364            }
365        }
366
367        // 提取类声明：class Name
368        if let Ok(re) = regex::Regex::new(r"class\s+(\w+)") {
369            for cap in re.captures_iter(source) {
370                if let Some(name) = cap.get(1) {
371                    let name_str = name.as_str();
372                    if let Some(pos) = source.find(name_str) {
373                        let location =
374                            self.compute_location_from_offsets(source, pos, pos + name_str.len());
375                        locations.insert(name_str.to_string(), location);
376                    }
377                }
378            }
379        }
380
381        locations
382    }
